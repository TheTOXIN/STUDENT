Алгоритм

 

  L - Удаление пользователя от головной части

  Lk - Длина кадра

  V - Время распространения сигнала

  I - Интенсивность передачи

  g:=1000;

  p:=v/l;

  a:=0.1;

 

  t:=l*v;

  m:=t/a;

  c:=lk/m;

  q:=0.36;

  q:=q*100;

 

  repeat

    x:=random(100)+1;

    if (q>x) then begin

       t1:=t1+m+t+2*t;

       y:=y+1;

       w:=0;

    end

    else begin

       w:=w+2;

       t1:=t1+w*t;

    end;

  until (y=q);

 

  v:=(g*(m+t+2*t))/t1;
