node* toList(node* head, int num)
{
    node* pv=new node;
     
    pv->num=num;
    pv->next=0;
    if (head)
    {
        node* cur=0;
        if(head->num < pv->num)
        {
            pv->next=head;
            head=pv;
        }
        else
        {
        cur=head;
        while((cur->next)&& (cur->next->num > pv->num))
          cur=cur->next;
        pv->next=cur->next;
        cur->next=pv;
        }
    }
    else
        head=pv;
    return head;
}
