#include "stdafx.h"
#include <iostream>
#include <string>
#include <math.h>
#include <locale.h>
#include <windows.h>
#include <fstream>
#include <stack>

using namespace std;

stack <char> s;

void input()
{
	char in;
	int  count;
	
	cout << "Введите размер:" << endl;
	cin >> count;

	cout << "Добавте элементы в стек" << endl;
	for (int i = 0; i < count; i++)
	{
		cin >> in;
		s.push(in);
	}
	cout << endl;
}

void print()
{ 
	stack <char> tmp = s;
	cout << "Стек: " << endl;
	int count = tmp.size();
	for (int i = 0; i < count; i++)
	{
		char c = tmp.top();
		cout << c << endl;
		tmp.pop();
	}
	cout << endl;
}

void delet()
{
	stack<char> tmp;
	int max = 0;

	for (int i = 0; i < s.size(); i++)
	{
		if (s.top() > max) 
		{
			max = s.top();
		}
	}

	cout << "Удаление элемента " << endl;
	
	int count = s.size();
	for (int i = 0; i < count; i++)
	{
		if (s.top() != max)
		{
			tmp.push(s.top());
		}
		s.pop();
	}

	s = tmp;
}

int main()
{
	SetConsoleCP(1251);
	SetConsoleOutputCP(1251);

	input();
	print();
	delet();
	print();

	system("pause");
	return 0;
}
