#include <iostream>
#include <string>
#include <math.h>
#include <locale.h>
#include <windows.h>
#include <fstream>
#include <stack>

using namespace std;

stack <char> s;//основной стек

void input()//функция ввода стека
{
	char in;
	int  count;

	cout << "Введите размер:" << endl;
	cin >> count;

	cout << "Добавте элементы в стек" << endl;
	for (int i = 0; i < count; i++)
	{
		cin >> in;
		s.push(in);//добавление элемента в стек
	}
	cout << endl;
}

void print()//функция печати стека
{
	stack <char> tmp = s;
	cout << "Стек: " << endl;
	int count = tmp.size();//размер стека
	for (int i = 0; i < count; i++)
	{
		char c = tmp.top();//смотрим элемент стека
		cout << c << endl;
		tmp.pop();//убираем его
	}
	cout << endl;
}

void delet()//функция удаления каждого второго элемента
{
	stack <char> tmp;//временный стек
	int count = s.size();//размер стека
	for (int i = 0; i < count; i++)
	{
		if (i % 2 == 0) {
			tmp.push(s.top());//кидаем в временный стек элемент элемент основного
			s.pop();//удаляем
		}
		else {
			s.pop();//в дргуих случаях мы просто убираем элемент с основного
		}
	}
	cout << endl;
	s = tmp;//заменяем основной на временный
}

int main()
{
	SetConsoleCP(1251);
	SetConsoleOutputCP(1251);

	input();
	print();
	delet();
	print();

	system("pause");
	return 0;
}
