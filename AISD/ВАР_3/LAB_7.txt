using namespace std;

struct Date//структура описывающаю дату
{
	int day;//день
	int mounth;//месяц
	int year;//год
};

class Order//класс описывающий заказ
{
public:
	int number;//номер заказа
	string mark;//марка
	string name;//имя
	Date last;//дата последниго ремонта
	Date end;//дата конца ремоннта

	Order() {

	}

	Order(string mark, string name)//конуструктор для инициализации данных класса
	{
		this->mark = mark;
		this->name = name;
		this->number = rand() % 100;
	}
};

Order orders[1000];//массив заказов
int len = 0;//кол-во заказов

{

	{

}

Date randomtDate()//функция которая возращает случаюную дату
{
	Date date;

	date.day = rand() % 31 + 1;
	date.mounth = rand() % 12 + 1;
	date.year = rand() % 18 + 2000;

	return date;
}

{
	inputSize();

		string name, mark;

		l = randomtDate();
		e = randomtDate();

		tmp.end = e;
		tmp.last = l;

	}
}

void randomOrder()//функция для случайного заполенения массива заказов
{
	inputSize();

	for (int i = 0; i < len; i++)
	{
		string name, mark;
		Date l, e;

		name = "Имя_" + to_string(rand() % len);
		mark = "Марка_" + to_string(rand() % len);

		if (rand() % 2 == 1)
			mark = "Жигули";

		l = randomtDate();
		e = randomtDate();

		Order tmp(mark, name);
		tmp.end = e;
		tmp.last = l;

		orders[i] = tmp;
	}
	system("pause");
}

int compareDate(Date d1, Date d2)//функция для сравнения двух дат (если d1 > d2 возрщает 1 иначе -1)
{
	if (d1.year > d2.year)
		return 1;
	else if (d1.year == d2.year)
	{
		if (d1.mounth > d2.mounth)
			return 1;
		else if (d1.mounth == d2.mounth) {
			if (d1.day < d2.day)
				return 1;
		}
	}

	return -1;
}

void sortOrder()//функция сортировки заказов(взято из премеров с сайта)
{
	cout << "Сортировка..." << endl;

	int i, j;
	Order k;

	for (j = 1; j < len; j++)
	{
		k = orders[j];
		i = j - 1;
		while (compareDate(k.last, orders[i].last) == 1 && i >= 0)
		{
			orders[i + 1] = orders[i];
			i -= 1;
		}
		orders[i + 1] = k;
	}

	system("pause");
}

}
