#include<iostream>
#include<string>
#include<conio.h>
#include<time.h>
#include<windows.h>

using namespace std;

string list[1000];//список билетов
int len = 0;//кол-во билетов

void showMenu()//функция вывода меню
{
	system("cls");
	cout << "===Выберети пункт===" << endl;
	cout << "1.Вывести билеты" << endl;
	cout << "2.Ввести билеты" << endl;
	cout << "3.Заполнить случайно" << endl;
	cout << "4.Сортировать" << endl;
	cout << "5.Показать счастливые билеты" << endl;
	cout << "6.Выход" << endl;
}

void inputSize()//функция ввода кол-во билетов
{
	cout << "Введите кол-во номеров: ";
	cin >> len;
}

boolean checkTicket(string ticket)//фуннкция для проверки билета
{
	int sum1 = 0, sum2 = 0;

	for (int i = 0; i < 3; i++)
		sum1 += ticket[i];//считаем сумму 1 части
	for (int i = 3; i < 6; i++)
		sum2 += ticket[i];//считаем сумму 2 части

	if (sum1 == sum2)//возращаем true если они равны
		return true;
	else
		return false;
}

void outputList(bool isHappy)//функция вывода билетов
{
	cout << "---НОМЕРА БИЛЕТОВ---" << endl;

	for (int i = 0; i < len; i++)
	{
		if (isHappy && checkTicket(list[i]))//если нам нужны счастливые билеты и билет счастливый то выводим его
			cout << i + 1 << "# = " << list[i] << endl;

		if (!isHappy)//если нам нужны все белеты то просто выводим их
			cout << i + 1 << "# = " << list[i] << endl;
	}
	cout << endl;

	system("pause");
}

void inputList()//функция ввода билетов
{
	inputSize();
	cout << "Введите 6-х значные номера билетов: " << endl;
	for (int i = 0; i < len; i++)
	{
		cout << i + 1 << "# ";
		string in;
		do
		{
			cin >> in;
		} while (in.length() != 6);//вводим билет пока он не 6-значный
	}
}

void randomList()//функция для заполенния случайными билетами
{
	inputSize();

	for (int i = 0; i < len; i++)
	{
		int tmp = rand() % 90000 + 99999;
		list[i] = to_string(tmp);
	}

	system("pause");
}

void sortList()//функция для сортировки методом пузырька
{
	cout << "Сортировка..." << endl;
	//подробнее тут :)
	//https://www.youtube.com/watch?v=7vqkHGKbb8k
	for (int i = 0; i < len; i++)
	{
		for (int j = len - 1; j > i; j--)
		{
			if (list[j] < list[j - 1])
			{
				string tmp = list[j];
				list[j] = list[j - 1];
				list[j - 1] = tmp;
			}
		}
	}

	system("pause");
}

int main()
{
	srand(time(NULL));

	SetConsoleCP(1251);
	SetConsoleOutputCP(1251);
	//массив значений по умолчанию
	int def[10] = { 124512, 342351, 765891, 453122, 431350, 876432, 734626, 238651, 455734, 234987 };
	len = 10;

	for (int i = 0; i < 15; i++)//добавляем значения по умолчанию
	{
		list[i] = to_string(def[i]);//в наш массив
	}
	//логика меню
	int choose = 0;
	while (choose != 6)
	{
		showMenu();
		cin >> choose;
		switch (choose)
		{
		case 1: outputList(false); break;
		case 2: inputList(); break;
		case 3: randomList(); break;
		case 4: sortList(); break;
		case 5: outputList(true);
		default: cout << "Ошибка ввода" << endl; break;
		}
	}

	system("pause");
	return 1;
}
