#include "stdafx.h"
#include <iostream>
#include <math.h>
#include <Windows.h>

using namespace std;

int main()
{

	double e = 0.001, s = 0, sum = 0, w = 0;
	int fact = 1;

	for (int i = 1; i <= 20; i++)//создаём цикл на 20 повторений для внешней сигмы
	{
		s = 0;//здесь мы обнуляем переменную s
		do
		{
			int j = 1;
			for (int k = 1; k <= j; k++)//через этот цикл считается факториал числа j
				fact = fact * k;
			s = sqrt(i + j) / fact;//формула решение котороый складывается в переменную sum 
			sum = sum + s;
			j++;
		} 
		while (fabs(s) < e);//цикл do while выполняется пока истенна данное условие 
		w = w + sum * log(i);
	}
	w = w * 2.7;
	cout << w << endl;
	system("pause");
	return 0;
}

#include "stdafx.h"
#include <iostream>
#include <math.h>
#include <Windows.h>

using namespace std;

int main()
{
	double Z = 0, max = 0, min = 0;
	double y = 0, dy = 0.5;
	double x = 0, dx = 0.2;
	do//цикл по x от 0 до 1 с увеличением на 0.2 каждый шаг
	{
		y = 0;
		do//цикл по y от 0 до 2.0 с увеличением на 0.5 каждый шаг
		{       
			//проверка условий для нахождения Z
			if (x > 0 && y < 0.5)
				Z = pow(x, 2) + y;
			else if (x < y && x > 0.5)
				Z = sqrt(pow(y, 2) + sin(x));
			else if (x > y && x >= 0.5)
				Z = pow(x, 2) + cos(y);
			//если это первый шаг в цикле по x то присваиваем min и max значение z 
			//чтобы было с чем сравнивать другие решения z для нахождения минимального и максимального z
			if (x == 0)
			{
				min = Z;
				max = Z;
			}
			//проверка на максимальность и минимальность
			if (Z > max)
				max = Z;
			if (Z < min)
				min = Z;
			cout << "Z = " << Z << endl;
			y += dy;
		} 
		while (y <= 2.0);
	
		x += dx;
	} 
	while (x <= 1.0);

	cout << "Z max = " << max << endl;
	cout << "Z min = " << min << endl;

	system("pause");
	return 0;
}
