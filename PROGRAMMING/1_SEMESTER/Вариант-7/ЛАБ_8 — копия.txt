#include "stdafx.h"
#include <iostream>
#include <math.h>
#include <locale.h>
#include <Windows.h>

using namespace std;

void operation(int n)//функция opearation принимает число n она выполняется после её вызова
{
	double rez1 = 0;
	double rez2 = 0;

	for (int a = 1; a <= n; a++)//через вложенный цикл перебираем все возможные коомбинации a и b
	{
		for (int b = 1; b <= n; b++)
		{
			rez1 = a - b + a % b;//заносим результат операции в переменную rez1
			rez2 = b - a + b % a;//делаем тоже самое для rez2 но a и b меняем местами

			if (rez1 == rez2)//если наши rez1 и rez2 равно ты воводим на экран a и b
				cout << "a = " << a << "  " << "b = " << b << endl;
		}
	}
}

int main()//программа начинает всегда выполняться с функции main
{
	setlocale(0, "RUS");

	int n;
	cout << "Введите число n: " << endl;
	cin >> n;
	cout << "Пары a и b не привосходящие n: " << endl;
	operation(n);//вызываем функцию operation и передаем туда число n

	system("pause");
	return 0;
}

void function_2(int m[10])//функция для вывода массива m
{

	{
		cout << m[i] << " ";

}

void function_3(int m[10])//функция для поиска и замены максимального и минимального значения
{
	int max;
	int min;
	int i1;
	int i2;

	for (int i = 0; i < 10; i++)//цикл для поиска max и min значения
	{
		if (i == 0)//если это первый шаг функции то приравниваем max и min чтобы было с чем сравнивать
		{
			max = m[i];
			min = m[i];
		}
		if (m[i] >= max)//ищем максимально значение
		{
			max = m[i];
			i1 = i;//добавляем в переменную i1 индекс максимального значния для будущей земены
		}
		if (m[i] <= min)//ищем минимальное значение
		{
			min = m[i];
			i2 = i;//добавляем в переменную i2 индекс минимального значния для будущей земены
		}
	}

	for (int i = 0; i < 10; i++)//цикл для замены max и min местами в массиве
	{
		if (i1 == i)
			m[i] = min;
		if (i2 == i)
			m[i] = max;
	}
}
//после завершения работы функции программа продолжает выполняться с момента её вызова
