#include "stdafx.h"
#include <iostream>
#include <math.h>
#include <locale.h>
#include <Windows.h>

using namespace std;

int main()
{
	setlocale(0, "RUS");
	
	int m[10];
	cout << "Введите элементы вашего массива: " << endl;
	for (int i = 0; i < 10; i++)
	{
		cout << "Введите элемент номер - " << i << " : ";
		cin >> m[i];
	}
	cout << "Индексы элементов, значения которых больше значения предыдущего элемента: " << endl;
	for (int i = 1; i < 10; i++)
	{
		if (m[i] > m[i - 1])
			cout << i << " ";
	}
	cout << endl;

	system("pause");
	return 0;
}

#include "stdafx.h"
#include <iostream>
#include <math.h>
#include <locale.h>
#include <Windows.h>
#include <time.h>

using namespace std;

int main()
{
	setlocale(0, "RUS");
	srand((unsigned)time(NULL));
	
	int A[4][4];
	int B[4];

	cout << "Массив A: " << endl;
	for (int i = 0; i < 4; i++)
	{
		for (int j = 0; j < 4; j++)
		{
			A[i][j] = rand() % 18 - 9;
			cout << A[i][j] << " ";
		}
		cout << endl;
	}

	cout << "Массив B: " << endl;
	for (int i = 0; i < 4; i++)
	{
		for (int j = 0; j < 4; j++)
		{
			if (A[i][j] < 0)
			{
				B[j] = j;
				cout << B[j] << " ";
				break;
			}
		}
	}

	system("pause");
	return 0;
}
