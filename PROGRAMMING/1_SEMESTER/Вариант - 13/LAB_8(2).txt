#include <iostream>
#include <math.h>

#pragma argsused

void function_1(int m[], int n)
{
	for (int i = 0; i < n; i++)
		cin >> m[i];
	cout << endl;
}

int function_2(int m[], int n)
{
	double x = 0;

	for (int i = 0; i < n; i++)
		x = x + sqrt(pow(m[i], 2));

	return x;
}

int main()
{
	const int n = 5;
	int m1[n];
	int m2[n];

	cout << "Input array 1: " << endl;
	function_1(m1, n);
	cout << "Your array 1: " << endl;
	for (int i = 0; i < n; i++)
		cout << m1[i] << " ";
	cout << endl;

	cout << "Input array 2 : " << endl;
	function_1(m2, n);
	cout << "Your array 2: " << endl;
	for (int i = 0; i < n; i++)
		cout << m2[i] << " ";
	cout << endl;

	cout << "Summ length array_1 and array_2: " << endl;
	cout << fabs(function_2(m1, n) + function_2(m2,n)) << endl;

	getch();
	return 0;
}
