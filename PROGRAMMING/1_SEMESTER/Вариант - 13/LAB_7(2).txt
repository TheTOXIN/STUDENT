#include <iostream>
#include <time.h>

#pragma argsused

int main()
{
	srand((unsigned)time(NULL));

	const int m = 3;
	int M[m][m];

	cout << "Your matrix:" << endl;
	for (int i = 0; i < m; i++)
	{
		for (int j = 0; j < m; j++)
		{
			M[i][j] = rand() % 2;
			cout << M[i][j] << " ";
		}
		cout << endl;
	}

	cout << "Numbers rows without 1 : ";
	bool b = false;
	for (int i = 0; i < m; i++)
	{
		int count = 0;

		for (int j = 0; j < m; j++)
		{
			if (M[i][j] == 0)
				count++;
		}

		if (count == m)
		{
			cout << i << " ";
			b = true;
		}
	}
	if (b == false)
		cout << "No such rows" << endl;
	else
		cout << endl;

	cout << "Numbers columns without 1 : ";
	for (int i = 0; i < m; i++)
	{
		int count = 0;

		for (int j = 0; j < m; j++)
		{
			if (M[j][i] == 0)
				count++;
		}

		if (count == m)
		{
			cout << i << " ";
			b = true;
		}
		
	}
	if (b == false)
		cout << "No such columns" << endl;
	else
		cout << endl;

	getch();
	return 0;
}
