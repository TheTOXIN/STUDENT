using namespace std;

void func_1(int * N, int n)//функция для заполнеиня и вывода массива, принимает массив и его размер
{
	for (int i = 0; i < n; i++)//заполняем массив случайными числами через цикл
	{
		N[i] = rand() % 100;//заполняем случайными числаом от 0 до 99
		cout << N[i] << " ";//выводим его
	}
	cout << endl;
}

int func_2(int* N, int n)//функция для нахождения количсеива нужных нам чисел в массиве, принимает сам массив и размер
{
	//эти переменный будут хранить
	int count = 0;//количестов нужных на чисел
	int namber;//число из массива
	int a;//цифру из чсила для сравнения
	int tmp;//последнию цифру их числа
	bool b = false;//истину если a == tmp, иначе ложь;

	for (int i = 0; i < n; i++)//обходим весь массив с числами
	{
		namber = N[i];//добавляем число из массива в namber
		a = namber % 10;//добавляем последнию цифру из числа для сравнения
		do//запускаем цикл пока namber не будет равен 0
		{
			tmp = namber % 10;//помещаем в tmp послдению цифру из числа
			namber = namber / 10;//делим число на 10 чтобы убрать еденицы
			if (tmp == a)//если цифра числа равна с цифрой для сравнения a то b = true
			{
				b = true;
			}
			else//иначе b = false, это значит что в нашем числе разные цифры и выходим из цикла
			{
				b = false;
				break;//выходим из цикла
			}
		} while (namber != 0);
		if (b == true)//если b отсался true увеличиваем count
		{
			count++;
		}
	}
	return count; возращаем count в место вызова функции
}

int main()
{
	setlocale(LC_ALL, "Rus");
	srand((unsigned)time(NULL));//для функции rand()

	int n;//будет содержать размер массива
	cout << "Введите размер массива: ";
	cin >> n;//вводим размер 
	int* N = new int[n];//создаем динамичесикй массив N с размером n
	func_1(N, n);//вызываем функция для заполнения массива передаем туда сам массив и его размер
	int count = func_2(N, n);//вызываем функция для поиска количсетва нужныйх чисел в массиве, передаем туда сам массив и его размер, и помещаем в count то что врент функция
	cout << "Количество чисел с одинаковыми цифрами = " << count << endl;//выводим count

	system("pause");
	return 0;
}
