using namespace std;

int** maker(int m, int n)
{
	int** t = new int*[m];
	for (int i = 0; i < m; i++)
	{
		t[i] = new int[n];
		for (int j = 0; j < n; j++)
		{
			t[i][j] = rand() % 10;
		}
	}
	return t;
}

void printer(int m, int n, int **matr)
{
	for (int i = 0; i < m; i++)
	{
		for (int j = 0; j < n; j++)
		{
			cout.width(4);
			cout << matr[i][j];
		}
		cout << endl;
	}
}

int main()
{
	setlocale(LC_ALL, "Rus");
	srand(time(NULL));

	int m, n;
	cout << "Введите размер матриц M x N: " << endl;
	cin >> m >> n;

	cout << "Матрица А :" << endl;
	int **A = maker(m, n);
	printer(m, n, A);

	for (int i = 0; i < n; i++)
	{
		for (int j = 0; j < m; j++)
		{
			if (A[j][i] % 2 != 0)
				break;
			if (j == m - 1)
				cout << "Cтолбце номер " << i + 1 << " четный" << endl;
		}
	}

	for (int i = 0; i < m; i++)
		delete[] A[i];
	delete[] A;

	system("pause");
	return 0;
}
