#include "stdafx.h"
#include <iostream>
#include <string>
#include <locale>
#include <windows.h>

using namespace std;

int main() {

	SetConsoleCP(1251);
	SetConsoleOutputCP(1251);

	string name;
	string chars;

	cout << "Введите ваше имя:" << endl;
	cin >> name;

	cout << "Введите символы:" << endl;
	cin >> chars;

	cout << "Кол-во букв имени из множества символов:" << endl;
	int count;
	for (int i = 0; i < name.length(); i++) 
	{
		count = 0;
		for (int j = 0; j < chars.length(); j++)
		{
			if (name[i] == chars[j])
			{
				count++;
			}
		}
		cout << name[i] << " - " << count << endl;
	}

	system("pause");
	return 0;
}
