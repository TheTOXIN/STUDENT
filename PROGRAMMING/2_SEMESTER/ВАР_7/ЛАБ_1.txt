#include <iostream>
#include <math.h>
#include <locale.h>
#include <string.h>
#include <stdlib.h>
#include <ctype.h>

using namespace std;

int main()
{
	setlocale(LC_ALL, "1251");

	const int n = 100, kol = 10;
	typedef char String[n];
	String words[kol];

	cout << "Input 10 words: " << endl;
	for (int i = 0; i < 10; i++)
		cin >> words[i];

	cout << "Your words: " << endl;
	for (int i = 0; i < 10; i++)
		cout << i + 1 << " - " << words[i] << endl;

	cout << "Your unique word: " << endl;

	bool b;
	int num = 0;
	char *max_word;
	char *me_word;

	for (int i = 0; i < 10; i++)
	{
		if (i == 0)
			max_word = words[i];
		if (strlen(words[i]) >= strlen(max_word))
			max_word = words[i];
	}

	for (int i = 0; i < 10; i++)
	{
		char* word = { words[i] };
		b = true;

		for (int j = 0; j < strlen(word)-1; j++)
		{
			for (int k = j + 1; k < strlen(word); k++)
			{
				if (word[j] == word[k])
					b = false;
			}
		}
		if (b == true && max_word == words[i])
			me_word = words[i];
	}
	cout << me_word << endl;

	system("pause");
	return 0;
}
