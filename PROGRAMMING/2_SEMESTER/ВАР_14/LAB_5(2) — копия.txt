#include "stdafx.h"
#include <iostream>
#include <math.h>
#include <locale.h>
#include "string"
#include <windows.h>
#include <fstream>

using namespace std;

const int n = 3;//кол-во строк
double A[n];//массив для 1 файла
double B[n];//массив для второго файла

void input(double m[], string name)//функция ввода
{
	ofstream fout(name, ios::binary);//открываем поток для записи в файл
	cout << "Введите компоненты файла " << name << endl;
	for (int i = 0; i < n; i++)//вводим текст
	{
		cin >> m[i];
		fout.write((char*)m, sizeof(m));
	}
	fout.close();
}

void print(double m[], string name)//функция для вывода файла
{
	ifstream fin(name, ios::binary);//открываем поток для чтения файла
	cout << "Компоненты файла " << name << endl;
	for (int i = 0; i < n; i++)//считываем файлы в массив и выводим
	{
		fin.read((char*)m, sizeof(m));
		cout << m[i] << endl;
	}
	fin.close();
}

void check()//функция для проверки файлов
{
	for (int i = 0; i < n; i++)//пробегаемся по массивам
	{
		if (A[i] != B[i])//если они не равны выходим из цикла
		{
			cout << "Компоненты файлов не равны" << endl;
			break;
		}

		if (i == n-1)//если последний элементы равны  и мы не вышли их цикла
		{
			cout << "Компоненты файлов равны" << endl;
		}
	}
}

int main()
{
	SetConsoleCP(1251);//ставим русский язык в консоль
	SetConsoleOutputCP(1251);

	input(A, "A.dat");
	input(B, "B.dat");
	print(A, "A.dat");
	print(B, "B.dat");
	check();

	system("pause");
	return 0;
}
