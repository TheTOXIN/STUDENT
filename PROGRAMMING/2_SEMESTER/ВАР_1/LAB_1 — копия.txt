#include "stdafx.h"
#include <iostream>
#include <math.h>
#include <locale.h>
#include <string.h>
#include <stdlib.h>
#include <ctype.h>
#include <windows.h>

using namespace std;

int main()
{
	//setlocale(LC_ALL, "Russian_Russia.866");
	const int n = 100, kol = 100;
	int count;
	typedef char String[n];//создаем тип String(строка)

	String words[kol];//делаем массив строк

	cout << "Input count words: " << endl;
	cin >> count;//вводим кол.во строк
	count++;

	cout << "Input words: " << endl;
	for (int i = 0; i < count; i++)//цикл для ввода массива строк
	{
		cin.getline(words[i], n);
	}
	cout << endl;

	cout << "List of words: ";//цикл для вывода массива строк
	for (int i = 0; i < count; i++)
	{
		cout << words[i] << endl;
	}
	cout << endl;

	cout << "List of correct words: " << endl;
	for (int i = 0; i < count; i++)//пробегаемся по массиву строк через цикл
	{
		int len = strlen(words[i]);//находим длину пробегаемого слова

		for (int j = 0; j < len - 1; j++)//через цикл пробегаемся по символам в нашем слове
		{
			if (words[i][j] == 'z' || words[i][j] == 'h')//если символ равен ж или ш мы проверяем вторую букву
			{
				if (words[i][j+1] == 'u')//если вторая буква равна ы меняем на и
				{
					words[i][j+1] = 'e';
				}
			}
			//z-ж h-ш u-ы e-и c-ч j-щ q-ю y-у i-я a-a 
			if (words[i][j] == 'c' || words[i][j] == 'j')//тоже самое для букв ч и щ
			{
				if (words[i][j+1] == 'q')
				{
					words[i][j+1] = 'y';
				}
				if (words[i][j+1] == 'i')
				{
					words[i][j+1] = 'a';
				}
			}
		}
	}

	for (int i = 0; i < count; i++)//цикл для вывода массива строк
	{
		cout << words[i] << endl;
	}
	cout << endl;

	system("pause");
	return 0;
}
